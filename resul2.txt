	function autoCompleteDaum(){
		var search_val = $(".in_keyword").val().replace(/(^\s*)|(\s*$)/g,"");
		var cnt = search_val.length;

		if(cnt >= 1){
			$(".daumsearch-layer").show();
			var data = { "q" : search_val};
			var li_search = "";

			$.ajax({
				type:"GET",
				url:"http://apihub.daum.net/suggest-affiliate/v2/mobile.json?callback=?&appkey=f70b7b4a8ff27b96c82c8337efd2f8df",
				cache:false,
				data: data,
				dataType : "jsonp",
				success: function(data){
					var len =  data['subkeys'].length;
					if(len > 10){
						len = 10;
					}
					var highlight = data['q'];
					if(len > 0){
						for(i=0; i< len; i++){						
							var rawresult = data['subkeys'][i];
							
							if(rawresult){ 
								var result = rawresult.replace(new RegExp(highlight, 'gi'), "<strong>" + data['q'] + "</strong>");
							}
							var highlighted = data['highlighted'][i];
							li_search += "<li><a class='js-search' href='http://search.daum.net/search?&w=tot&DA=DIF&q="+rawresult+"' target = '_blank'>" + result + "</a><button class='btn_add'><span class='ico_add'>추가</span></button></li>";
						}
					} else {
						li_search += "";
					}
					$(".daumsearch-list").html(li_search);
				}
			});
		}
	}
